# Cron jobs for maddy-bot scheduled triggers
# Add with: crontab -e

# 1st of every month at 9:00 AM (ICT = UTC+7, so 02:00 UTC)
0 2 1 * * curl -s -X POST http://127.0.0.1:8000/trigger/monthly-allocation

# Every Sunday at 8:00 PM ICT (13:00 UTC)
0 13 * * 0 curl -s -X POST http://127.0.0.1:8000/trigger/weekly

# Last day of month at 9:00 PM ICT (14:00 UTC) — runs on day 28,29,30,31
# The handler itself checks whether today is actually the last day
0 14 28-31 * * curl -s -X POST http://127.0.0.1:8000/trigger/monthly-report
